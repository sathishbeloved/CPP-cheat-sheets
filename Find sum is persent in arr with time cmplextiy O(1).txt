bool calc(int arr[],int n,int k){
        int left = 0, right =sizeof(arr);
        std::cout << right << std::endl;
    while(left <= right){
 
        if(arr[left] + arr[right] == k)
        return true;
        else if(arr[left] + arr[right] > k)
        right--;
        else
        left++;
        
    }
    return false;
}
